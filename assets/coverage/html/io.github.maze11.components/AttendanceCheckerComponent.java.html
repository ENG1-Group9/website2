package io.github.maze11.components;

import com.badlogic.ashley.core.Component;

/** Component for Attendance Monitoring Functionality */
public class AttendanceCheckerComponent implements Component {
  public float attendance; // The current attendance value (as decimal)
  public float totalCodes; // The number of check in codes on the map
  public float collectedCodes; // The number of check in codes that the player has collected

  // No-argument constructor required by Ashley's PooledEngine
  public AttendanceCheckerComponent() {
    this.attendance = 0f;
    this.totalCodes = 13f; // should be updated in code after declared
    this.collectedCodes = 0f;
  }

  public void setTotalCodes(int newValue) {
    totalCodes = newValue;
  }

  /**
   * Cantculate attendance attendance = collectedCodes / toalCodes, rounded down
   *
   * @return attendance
   */
  public float calculateAttendance() {
    attendance = collectedCodes / totalCodes;
    return attendance;
  }

  /** Tell the attendance checker that a check in code has been collected */
  public void codeCollected() {
    if (collectedCodes < totalCodes) {
      collectedCodes++;
    }
    calculateAttendance();
  }

  public float getCollectedCodes() {
    return collectedCodes;
  }
}

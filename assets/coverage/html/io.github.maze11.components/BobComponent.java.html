package io.github.maze11.components;

import com.badlogic.ashley.core.Component;
import com.badlogic.ashley.core.ComponentMapper;
import com.badlogic.ashley.core.Entity;
import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.Texture;
import io.github.maze11.messages.Message;
import io.github.maze11.messages.MessagePublisher;
import io.github.maze11.messages.MessageType;
import io.github.maze11.messages.ToastMessage;

public class BobComponent implements Component {
  private boolean hasGlasses = false; // Wether Bob has the glasses
  private boolean glassesCollected = false; // Wether the player has found the galasses

  private final ComponentMapper<SpriteComponent> spriteM =
      ComponentMapper.getFor(SpriteComponent.class);
  private final ComponentMapper<InteractableComponent> intM =
      ComponentMapper.getFor(InteractableComponent.class);
  private Texture glassesTexture;

  public void init(Texture glassesTexture) {
    this.glassesTexture = glassesTexture;
  }

  public void giveGlasses(Entity bobEntity, MessagePublisher messagePublisher) {
    InteractableComponent interact = intM.get(bobEntity);
    interact.interactionEnabled = true;
    if (!hasGlasses && glassesCollected) {
      hasGlasses = true;
      spriteM.get(bobEntity).setTexture(glassesTexture);
      Color bobColour = new Color(0.071f, 0.381f, 0.126f, 0f);
      interact.additionalMessages.set(
          0, new ToastMessage("Bob: Thanks for the glasses!", 2f, bobColour));
      messagePublisher.publish(new ToastMessage("Bob: Hey you found my glasses!", 2f, bobColour));
      messagePublisher.publish(new Message(MessageType.GIVE_GLASSES));
    }
  }

  public void collectGlasses() {
    glassesCollected = true;
  }

  public boolean getHasGlasses() {
    return hasGlasses;
  }
}

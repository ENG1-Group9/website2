package io.github.maze11.screens;

import com.badlogic.gdx.graphics.Color;
import com.badlogic.gdx.graphics.g2d.BitmapFont;
import com.badlogic.gdx.scenes.scene2d.InputEvent;
import com.badlogic.gdx.scenes.scene2d.ui.Label;
import com.badlogic.gdx.scenes.scene2d.ui.Table;
import com.badlogic.gdx.scenes.scene2d.ui.TextButton;
import com.badlogic.gdx.scenes.scene2d.utils.ClickListener;
import io.github.maze11.MazeGame;
import io.github.maze11.leaderboard.LeaderboardEntry;
import io.github.maze11.leaderboard.LeaderboardManager;
import io.github.maze11.ui.FontGenerator;
import java.util.ArrayList;

public class LeaderboardScreen extends BaseMenuScreen {
  public LeaderboardScreen(MazeGame game) {
    super(game);
    buildUI();
  }

  @Override
  protected void buildUI() {
    Table table = new Table();
    table.setFillParent(true);

    Label.LabelStyle titleStyle = new Label.LabelStyle(titleFont, Color.WHITE);
    Label title = new Label("Leaderboard", titleStyle);

    Label.LabelStyle entryStyle = new Label.LabelStyle(bodyFont, Color.LIGHT_GRAY);

    table.add(title).padBottom(30).row();

    LeaderboardManager leaderboardManager = new LeaderboardManager();
    ArrayList<LeaderboardEntry> scores = leaderboardManager.getTopScores();

    if (scores.isEmpty()) {
      table
          .add(new Label("Beat the game to add scores to the leaderboard", entryStyle))
          .padBottom(20)
          .row();
    } else {
      Table scoreTable = new Table();
      scoreTable.add(new Label("Rank", entryStyle)).padRight(20);
      scoreTable.add(new Label("Name", entryStyle)).padRight(50);
      scoreTable.add(new Label("Score", entryStyle)).row();

      int rank = 1;
      for (LeaderboardEntry entry : scores) {
        scoreTable.add(new Label(String.valueOf(rank), entryStyle)).padRight(20);
        scoreTable.add(new Label(entry.getName(), entryStyle)).padRight(50);
        scoreTable.add(new Label(String.valueOf(entry.getScore()), entryStyle)).row();
        rank++;
      }
      table.add(scoreTable).padBottom(30).row();
    }

    BitmapFont buttonFont = FontGenerator.generateRobotoFont(24, Color.WHITE, skin);
    TextButton.TextButtonStyle buttonStyle =
        new TextButton.TextButtonStyle(skin.get(TextButton.TextButtonStyle.class));
    buttonStyle.font = buttonFont;

    TextButton backButton = new TextButton("Back", buttonStyle);
    backButton.addListener(
        new ClickListener() {
          @Override
          public void clicked(InputEvent event, float x, float y) {
            game.switchScreen(new MenuScreen(game));
            playMenuClick();
          }
        });

    table.add(backButton).width(200).height(60);

    stage.addActor(table);
  }

  @Override
  protected float[] getBackgroundColor() {
    return new float[] {0.1f, 0.1f, 0.2f}; // Dark blue
  }
}

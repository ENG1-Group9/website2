package io.github.maze11.messages;

import com.badlogic.gdx.graphics.Color;

/** Requests a toast to appear on the screen */
public class ToastMessage extends Message {

  /** The text to display on the toast */
  private final String text;

  /** The duration to display the toast for */
  private float durationLeft;

  private Color backColour;
  private Color textColour;

  public ToastMessage(String text, float duration) {
    super(MessageType.TOAST_SHOW);
    this.text = text;
    this.durationLeft = duration;
    this.backColour = new Color();
    backColour.set(0.1f, 0.1f, 0.1f, 0f);
    this.textColour = new Color();
    textColour.set(1f, 1f, 0.9f, 0);
  }

  public ToastMessage(String text, float duration, Color backColour, Color textColour) {
    super(MessageType.TOAST_SHOW);
    this.text = text;
    this.durationLeft = duration;
    this.backColour = backColour;
    this.textColour = textColour;
  }

  public ToastMessage(String text, float duration, Color backColour) {
    super(MessageType.TOAST_SHOW);
    this.text = text;
    this.durationLeft = duration;
    this.backColour = backColour;
    this.textColour = new Color();
    textColour.set(1f, 1f, 0.9f, 0);
  }

  /** Update the duration the toast will stay on screen */
  public void setDuration(float duration) {
    this.durationLeft = duration;
  }

  /**
   * Decrease the duration left on this toast.
   *
   * @param amount
   */
  public void lowerDuration(float amount) {
    this.durationLeft -= amount;
  }

  public float getDurationLeft() {
    return durationLeft;
  }

  public String getText() {
    return text;
  }

  public Color getBackColour() {
    return backColour;
  }

  public Color getTextColour() {
    return textColour;
  }
}

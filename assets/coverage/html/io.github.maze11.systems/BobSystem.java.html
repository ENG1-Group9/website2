package io.github.maze11.systems;

import com.badlogic.ashley.core.ComponentMapper;
import com.badlogic.ashley.core.Entity;
import com.badlogic.ashley.core.Family;
import io.github.maze11.components.BobComponent;
import io.github.maze11.fixedStep.FixedStepper;
import io.github.maze11.fixedStep.IteratingFixedStepSystem;
import io.github.maze11.messages.Message;
import io.github.maze11.messages.MessageListener;
import io.github.maze11.messages.MessagePublisher;
import io.github.maze11.messages.MessageType;

public class BobSystem extends IteratingFixedStepSystem {
  private final MessageListener messageListener;
  private final MessagePublisher messagePublisher;
  private final ComponentMapper<BobComponent> bobM = ComponentMapper.getFor(BobComponent.class);

  public BobSystem(FixedStepper stepper, MessagePublisher messagePublisher) {
    super(stepper, Family.all(BobComponent.class).get());
    this.messagePublisher = messagePublisher;
    this.messageListener = new MessageListener(messagePublisher);
  }

  @Override
  public void fixedUpdate(float deltaTime) {
    while (messageListener.hasNext()) {
      Message msg = messageListener.next();
      if (msg.type == MessageType.COLLECT_GLASSES) {
        System.out.println("i");
        for (Entity entity : getEntities()) {
          BobComponent bob = bobM.get(entity);
          bob.collectGlasses();
        }
      }
      if (msg.type == MessageType.TALK_BOB) {
        for (Entity entity : getEntities()) {
          BobComponent bob = bobM.get(entity);
          bob.giveGlasses(entity, messagePublisher);
        }
      }
    }
  }

  @Override
  protected void fixedStepProcessEntity(Entity entity, float deltaTime) {
    // No actions need to be taken
  }
}

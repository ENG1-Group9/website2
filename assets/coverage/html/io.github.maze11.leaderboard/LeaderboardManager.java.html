package io.github.maze11.leaderboard;

import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Preferences;
import com.badlogic.gdx.utils.Json;
import com.badlogic.gdx.utils.JsonWriter;
import java.util.ArrayList;
import java.util.Collections;

// json is needed due to limitations in the prefs system in libgdx i think

public class LeaderboardManager {
  private static final String PREF_NAME = "MazeGameLeaderboard";
  private static final String KEY_SCORES = "scores";
  private static final int MAX_SCORES = 5;

  private ArrayList<LeaderboardEntry> scores;
  private Preferences prefs;
  private Json json;

  public LeaderboardManager() {
    prefs = Gdx.app.getPreferences(PREF_NAME);
    json = new Json();
    json.setOutputType(JsonWriter.OutputType.json);
    loadScores();
  }

  @SuppressWarnings("unchecked")
  private void loadScores() {
    String scoresJson = prefs.getString(KEY_SCORES, "[]");
    scores = json.fromJson(ArrayList.class, LeaderboardEntry.class, scoresJson);
    if (scores == null) {
      scores = new ArrayList<>();
    }
    sortAndTrim();
  }

  private void saveScores() {
    String scoresJson = json.toJson(scores);
    prefs.putString(KEY_SCORES, scoresJson);
    prefs.flush();
  }

  private void sortAndTrim() {
    Collections.sort(scores);
    if (scores.size() > MAX_SCORES) {
      scores.subList(MAX_SCORES, scores.size()).clear();
    }
  }

  public void addScore(String name, int score) {
    scores.add(new LeaderboardEntry(name, score));
    sortAndTrim();
    saveScores();
  }

  public boolean isHighScore(int score) {
    if (scores.size() < MAX_SCORES) {
      return true;
    }
    return score > scores.get(scores.size() - 1).getScore();
  }

  public ArrayList<LeaderboardEntry> getTopScores() {
    return scores;
  }
}

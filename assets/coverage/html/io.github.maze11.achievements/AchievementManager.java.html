package io.github.maze11.achievements;

import java.util.HashMap;
import java.util.Map;

import com.badlogic.ashley.core.ComponentMapper;
import com.badlogic.ashley.core.Entity;
import com.badlogic.ashley.core.Family;
import com.badlogic.gdx.Gdx;
import com.badlogic.gdx.Preferences;
import com.badlogic.gdx.graphics.Color;

import io.github.maze11.components.AttendanceCheckerComponent;
import io.github.maze11.fixedStep.FixedStepper;
import io.github.maze11.fixedStep.IteratingFixedStepSystem;
import io.github.maze11.messages.Message;
import io.github.maze11.messages.MessageListener;
import io.github.maze11.messages.MessagePublisher;
import io.github.maze11.messages.ToastMessage;
import io.github.maze11.systems.PiSystem;

// currently achievements is not seen by the player only in the code
// we need to show a toast or something when a player gets an achievement
// and show all achievements somewhere

public class AchievementManager extends IteratingFixedStepSystem {
  public static final String PREF_NAME = "MazeGameAchievementData";

  private static Preferences prefs;
  private static MessagePublisher messagePublisher;
  private static MessageListener messageListener;
  private static final Color messageColour = new Color(0.2f, 0.5f, 1f, 0f);

  private final ComponentMapper<AttendanceCheckerComponent> attendM =
      ComponentMapper.getFor(AttendanceCheckerComponent.class);

  private static Map<String, Achievement> achievements = new HashMap<>();
  public static final String[] achievementNames = {
    "Coffee Addict", "SYS1 Trauma", "Model Student", "CLICK!?!", "Pi Collecter", "Bird Food"
  };
  public static final String[] achievementDescriptions = {
    "Drink all coffee on the map.",
    "Help out an old friend.",
    "Get 100% attendance.",
    "Unlock the secret room.",
    "Complete the rasberry pi.",
    "Get attacked by a goose 10 times."
  };

  private static int coffeesCollected = 0;
  private static int totalCoffeesInLevel =
      0; // The game will update this value as it creates the coffee's
  private static int pisCollected = 0;
  private static int totalPisInLevel = 5; // The game will update this value as it creates the pi's
  private static int gooseBites = 0;

  public AchievementManager(FixedStepper stepper, MessagePublisher messagePublisher) {
    super(stepper, Family.all(AttendanceCheckerComponent.class).get());
    AchievementManager.messagePublisher = messagePublisher;
    messageListener = new MessageListener(messagePublisher);

    prefs = Gdx.app.getPreferences(PREF_NAME);
    for (int i = 0; i < achievementNames.length; i++) {
      achievements.put(
          achievementNames[i], new Achievement(achievementNames[i], achievementDescriptions[i]));
    }
    loadScores();
  }

  public static void newRound() {
    // Just resets for the current game like if the player quits and plays again
    coffeesCollected = 0;
    pisCollected = 0;
    totalCoffeesInLevel = 0;
    totalCoffeesInLevel = 0;
    PiSystem.reset();
  }

  public void setMessagePublisher(MessagePublisher p) {
    messagePublisher = p;
    messageListener = new MessageListener(messagePublisher);
  }

  @Override
  public void fixedUpdate(float deltaTime) {
    while (messageListener.hasNext()) {
      Message msg = messageListener.next();
      switch (msg.type) {
        case COLLECT_COFFEE -> {
          coffeesCollected++;
          if (coffeesCollected == totalCoffeesInLevel) {
            unlockAchievement(achievementNames[0]);
          }
        }
        case GIVE_GLASSES -> unlockAchievement(achievementNames[1]);
        case CHECK_IN_CODE_COLLECT -> {
          for (Entity entity : getEntities()) {
            AttendanceCheckerComponent attendanceChecker = attendM.get(entity);
            if (attendanceChecker.calculateAttendance() == 1) {
              unlockAchievement(achievementNames[2]);
            }
          }
        }
        case PRESSURE_PLATE_TRIGGER -> unlockAchievement(achievementNames[3]);
        case PI_COLLECT -> {
          pisCollected++;
          if (pisCollected == totalPisInLevel) {
            unlockAchievement(achievementNames[4]);
          }
        }
        case GOOSE_BITE -> {
          gooseBites++;
          if (gooseBites == 10) {
            unlockAchievement(achievementNames[5]);
          }
        }
        default -> {}
      }
    }
  }

  @Override
  protected void fixedStepProcessEntity(Entity entity, float deltaTime) {
    // No actions need to be taken
  }

  public static int getTotalCoffeeValue() {
    return totalCoffeesInLevel;
  }

  public static void onCoffeeMade() {
    totalCoffeesInLevel++;
  }

  private static void loadScores() {
    for (Achievement a : achievements.values()) {
      boolean unlocked = prefs.getBoolean(a.getName(), false);
      a.setUnlocked(unlocked);
    }
  }

  public static void unlockAchievement(String name) {
    Achievement a = achievements.get(name);

    if (!a.isUnlocked()) {
      a.setUnlocked(true);
      String unlockText = "Achievement Unlocked: " + name + "!";
      messagePublisher.publish(new ToastMessage(unlockText, 10, messageColour));
    }

    prefs.putBoolean(name, true);
    prefs.flush();
  }
}
